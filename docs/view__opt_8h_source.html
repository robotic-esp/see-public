137 inline bool ViewOptHemi(Vector3f vobs, MatrixXf &P, Vector3f &vopt,
138  int &result) {
139  double minf, eps = 1e-3;
140  std::vector<double> lb, ub, v;
141  std::shared_ptr<float[]> C(new float[P.rows() * P.cols()]);
142 
143  Map<MatrixXf>(C.get(), P.rows(), P.cols()) = P;
144  nlopt::opt opt(nlopt::LD_SLSQP, 4);
145 
146  opt.add_inequality_constraint(ViewOptHemiQuadConst, NULL, eps);
147  for (int i = 0; i < P.cols(); i++) {
148  void *C_i = static_cast<void *>(C.get() + (i * 4));
149  opt.add_inequality_constraint(ViewOptHemiPointConst, C_i, eps);
150  }
151 
152  lb = {0, -HUGE_VAL, -HUGE_VAL, -HUGE_VAL};
153  ub = {1, HUGE_VAL, HUGE_VAL, HUGE_VAL};
154  v = {0, vobs(0), vobs(1), vobs(2)};
155 
156  opt.set_maxtime(0.01);
157  opt.set_xtol_abs(eps);
158  opt.set_lower_bounds(lb);
159  opt.set_upper_bounds(ub);
160  opt.set_min_objective(ViewOptHemiObjFunction, NULL);
161 
162  try {
163  if (!result) {
164  result = opt.optimize(v, minf);
165  }
166  if (v[0] > eps) {
167  vopt << v[1] / sqrt(v[0]), v[2] / sqrt(v[0]), v[3] / sqrt(v[0]);
168  vopt.normalize();
169  return true;
170  }
171  } catch (std::exception &e) {
172  }
173  return false;
174 }
