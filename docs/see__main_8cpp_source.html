67 bool ReachedNBV(SeeView view, SeeView nbv, NBVParams nbv_p, bool &first_view) {
68  float dst_off, ort_off;
69 
70  if (first_view) {
71  first_view = false;
72  return true;
73  }
74 
75  dst_off = (view.getVector3fMap() - nbv.getVector3fMap()).norm();
76  ort_off = view.getViewVector3fMap().dot(nbv.getViewVector3fMap());
77  ort_off /= view.getViewVector3fMap().norm() * nbv.getViewVector3fMap().norm();
78  ort_off = acos(fmin(fmax(ort_off, -1.0), 1.0));
79 
80  return dst_off < nbv_p.dst_thres && (ort_off * 180 / M_PI) < nbv_p.ort_thres;
81 }
