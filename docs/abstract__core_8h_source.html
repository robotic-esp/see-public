123 inline bool AbstractCore::ComputeK(AlgParams &alg) {
124 
125  if (alg.r && alg.rho) {
126  alg.k = ceil(((4.0 / 3.0) * M_PI * pow(alg.r, 3.0)) * alg.rho);
127  alg.eps = pow((3.0 * alg.r) / (2.0 * M_PI * alg.rho), 1.0 / 3.0);
128 
129  if (alg.k < 3) {
130  float r = alg.r;
131  int rho = alg.rho;
132  alg.rho = ceil(9.0 / (4.0 * M_PI * pow(r, 3.0)));
133  alg.r = pow(9.0 / (4.0 * M_PI * rho), 1.0 / 3.0);
134  return false;
135  }
136  return true;
137  }
138  return false;

161 inline bool AbstractCore::SetViewDistance(AlgParams &alg, SensorParams &sen) {
162  int pix = sen.pix_x * sen.pix_y;
163  float f_x = M_PI * sen.fov_x / 360.0;
164  float f_y = M_PI * sen.fov_y / 360.0;
165 
166  if (!alg.d && alg.rho && alg.r) {
167  // alg.d = pow((3.0 * pix) / (4.0 * alg.rho * tan(f_x) * tan(f_y)), 1.0
168  // / 3.0);
169  float a = pix / (12.0 * alg.rho * tan(f_x) * tan(f_y));
170  alg.d = sqrt(a - ((2 * pow(alg.r, 2)) / 3.0));
171  return true;
172  }
173  return false;
174 }

182 inline bool AbstractCore::SetDensity(AlgParams &alg, SensorParams &sen) {
183  int pix = sen.pix_x * sen.pix_y;
184  float f_x = M_PI * sen.fov_x / 360.0;
185  float f_y = M_PI * sen.fov_y / 360.0;
186 
187  if (!alg.rho && alg.d && alg.r) {
188  float a = 4 * tan(f_x) * tan(f_y) * (3 * pow(alg.d, 2) + 2 * pow(alg.r, 2));
189  alg.rho = pix / a;
190  return true;
191  }
192  return false;
