123 
124  if (alg.r && alg.rho) {

126  alg.eps = pow((6 * alg.r) / (4 * M_PI * alg.rho), 1.0 / 3.0);

180  float f_x = M_PI * sen.fov_x / 360.0;
181  float f_y = M_PI * sen.fov_y / 360.0;
182 
183  if (!alg.rho && alg.d) {
184  float a = 4 * tan(f_x) * tan(f_y) * (3 * pow(alg.d, 2) + 2 * pow(alg.r, 2));
