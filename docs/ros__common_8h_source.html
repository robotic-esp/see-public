39 void AffineTransformToSeeView(Eigen::Affine3f transform, SeeView &view,
40  int f_axis) {
41  Eigen::Vector3f axis(0, 0, 0);
42  axis(f_axis) = 1;
43 
44  view.getVector3fMap() = transform.translation();
45  view.getViewVector3fMap() = (transform.rotation() * axis).normalized();
46 }

54 void SeeViewToAffineTransform(SeeView view, Eigen::Affine3f &transform,
55  int f_axis) {
56  Eigen::Vector3f axis(0, 0, 0);
57  axis(f_axis) = 1;
58 
59  transform =
60  Eigen::Translation3f(view.getVector3fMap()) *
61  Eigen::Quaternionf::FromTwoVectors(axis, view.getViewVector3fMap());
62 }

170  void OnSensorMessage(const sensor_msgs::PointCloud2ConstPtr &cloud_in) {
171  Eigen::Affine3d tf_affine;
172  Eigen::Matrix4f tf_matrix;
173  tf::StampedTransform transform;
174  sensor_msgs::PointCloud2 cloud_tmp;
175 
176  if (!(tf_tl.frameExists(s_p.world_frame) &&
177  tf_tl.frameExists(cloud_in->header.frame_id))) {
178  ROS_ERROR("Coordinate frame initialisation failed.");
179  ros::shutdown();
180  }
181 
182  if (static_cast<int>(cloud_in->header.seq) > cloud_num) {
183  if (transformPointCloud(s_p.world_frame, *cloud_in, cloud_tmp, tf_tl)) {
184  tf_tl.lookupTransform(s_p.world_frame, cloud_in->header.frame_id,
185  cloud_in->header.stamp, transform);
186  tf::transformTFToEigen(transform, tf_affine);
187  AffineTransformToSeeView(tf_affine.cast<float>(), view, s_p.f_axis);
188  pcl::fromROSMsg(cloud_tmp, *cloud);
189 
190  has_cloud = true;
191  cloud_num = static_cast<int>(cloud_in->header.seq);
192  ros::param::set("/see/sen_cloud_num", cloud_num);
193  }
194  }
